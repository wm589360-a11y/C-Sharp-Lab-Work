Definition : Write a program to implement interface called Exam with a method Pass (int mark) that returns a boolean. Write another interface called Classify with a method Division (int average) which returns a String. Write a class called Result which implements both Exam and Classify. The Pass method should return true if the mark is greater than or equal to 50 else false. The Division method must return “First” when the parameter average is 60 or more, “Second” when average is 50 or more but below 60, “No division” when average is less than 50. C#


using System;

// Interface Exam
interface IExam
{
    bool Pass(int mark);
}

// Interface Classify
interface IClassify
{
    string Division(int average);
}

// Class Result implementing both interfaces
class Result : IExam, IClassify
{
    // Implement Pass method
    public bool Pass(int mark)
    {
        return mark >= 50;
    }

    // Implement Division method
    public string Division(int average)
    {
        if (average >= 60)
            return "First";
        else if (average >= 50)
            return "Second";
        else
            return "No division";
    }
}

// Main class to test
class Program
{
    static void Main(string[] args)
    {
        Result r = new Result();

        int mark = 55;
        int average = 58;

        Console.WriteLine("Pass Status: " + r.Pass(mark));
        Console.WriteLine("Division: " + r.Division(average));
    }
}


Program Output:

Pass Status: True
Division: Second
