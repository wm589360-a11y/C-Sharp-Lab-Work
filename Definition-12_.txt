#Write a C# Program to count positive and negative numbers using array.

using System;

class PositiveNegativeCount
{
    static void Main()
    {
        Console.Write("Enter the number of elements: ");
        int n = int.Parse(Console.ReadLine());

        int[] numbers = new int[n];
        int positiveCount = 0;
        int negativeCount = 0;
        int zeroCount = 0;

        for (int i = 0; i < n; i++)
        {
            Console.Write($"Enter element {i + 1}: ");
            numbers[i] = int.Parse(Console.ReadLine());
        }

        foreach (int num in numbers)
        {
            if (num > 0)
                positiveCount++;
            else if (num < 0)
                negativeCount++;
            else
                zeroCount++;
        }

        Console.WriteLine("\nResults:");
        Console.WriteLine("Positive numbers: " + positiveCount);
        Console.WriteLine("Negative numbers: " + negativeCount);
        Console.WriteLine("Zero numbers: " + zeroCount);
    }
}

#Write a C# program using if condition to count positive or negative numbers separately.

using System;

class Program
{
    static void Main()
    {
        int positiveCount = 0;
        int negativeCount = 0;
        int n;

        Console.Write("Enter how many numbers you want to input: ");

        // Validate first input
        while (!int.TryParse(Console.ReadLine(), out n) || n <= 0)
        {
            Console.Write("Please enter a valid positive integer: ");
        }

        for (int i = 1; i <= n; i++)
        {
            int number;
            Console.Write("Enter number " + i + ": ");

            // Validate each number input
            while (!int.TryParse(Console.ReadLine(), out number))
            {
                Console.Write("Invalid input. Please enter a valid integer: ");
            }

            if (number > 0)
            {
                positiveCount++;
            }
            else if (number < 0)
            {
                negativeCount++;
            }
        }

        Console.WriteLine("\nTotal Positive Numbers: " + positiveCount);
        Console.WriteLine("Total Negative Numbers: " + negativeCount);
    }

#Write a C# program to calculate sum and average.

using System;

class Program
{
    static void Main()
    {
        int n;
        double sum = 0;

        Console.Write("Enter how many numbers you want to input: ");
        
        while (!int.TryParse(Console.ReadLine(), out n) || n <= 0)
        {
            Console.Write("Please enter a valid positive integer: ");
        }

        for (int i = 1; i <= n; i++)
        {
            double number;
            Console.Write("Enter number " + i + ": ");

            while (!double.TryParse(Console.ReadLine(), out number))
            {
                Console.Write("Invalid input. Please enter a valid number: ");
            }

            sum += number;
        }

        double average = sum / n;

        Console.WriteLine("\nSum = " + sum);
        Console.WriteLine("Average = " + average);
    }
}

#Write a C# program using a loop to calculate total marks.

{
    static void Main()
    {
        int[] marks = new int[5];
        int total = 0;

        Console.WriteLine("Enter marks for 5 students:");

        // Input marks and calculate total using loop

            Console.Write("Student " + (i + 1) + ": ");
            marks[i] = Convert.ToInt32(Console.ReadLine());
            total += marks[i];  // Add each mark to total

        // Display total marks
        Console.WriteLine("\nTotal Marks = " + total);
    }
}

#Write a C# program to calculate and display the average.

using System;

class Program
{
    static void Main()
    {
        int[] marks = new int[5];
        int total = 0;
        double average;

        Console.WriteLine("Enter marks for 5 students:");

        // Input marks and calculate total
        for (int i = 0; i < marks.Length; i++)
        {
            Console.Write("Student " + (i + 1) + ": ");
            marks[i] = Convert.ToInt32(Console.ReadLine());
            total += marks[i];
        }

        // Calculate average
        average = (double)total / marks.Length;

        // Display result
        Console.WriteLine("\nTotal Marks = " + total);
        Console.WriteLine("Average Marks = " + average);
    }
}

Student 1: 90
Student 2: 95
Student 3: 85
Student 4: 89
Student 5: 99
